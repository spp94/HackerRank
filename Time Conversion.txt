import java.io.*;
import java.math.*;
import java.text.*;
import java.util.*;
import java.util.regex.*;

public class Solution {

    /*
     * Complete the timeConversion function below.
     */
    static String timeConversion(String s) {
        String zone = s.substring(s.length()-2);
        System.out.println(zone);
        s = s.substring(0,s.length()-2);
        String[] timeZone = s.split(":");
        int hrs = Integer.parseInt(timeZone[0]);
        int min = Integer.parseInt(timeZone[1]);
        int sec = Integer.parseInt(timeZone[2]);
        if(zone.equals("PM"))
        {
            if(hrs != 12)
            return ((hrs+12)%24)+s.substring(2);
            else
            return s;
        }
        else
        {
            if(hrs == 12)
             return "00"+s.substring(2);
            else
             return s;
        }
    }

    private static final Scanner scan = new Scanner(System.in);

    public static void main(String[] args) throws IOException {
        BufferedWriter bw = new BufferedWriter(new FileWriter(System.getenv("OUTPUT_PATH")));

        String s = scan.nextLine();

        String result = timeConversion(s);

        bw.write(result);
        bw.newLine();

        bw.close();
    }
}
