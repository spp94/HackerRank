

public static int B;
public static int H;
public static boolean flag;

static
{
    try
    {
        Scanner sc = new Scanner(System.in);
        B = sc.nextInt();
        H = sc.nextInt();
        if(B <= 0 || H <= 0)
        {
            flag = false;
    System.out.println("java.lang.Exception: Breadth and height must be positive");
        }
        else
        {
            flag = true;   
        }
    }
    catch(Exception ex)
    {
        throw ex;
    }
}
